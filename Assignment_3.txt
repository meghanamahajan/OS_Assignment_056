#!/bin/bash
#prime number

echo -e "enter the number\c"
read input

if [ $input -eq $input ] 2>/dev/null
then
        for((i=2;i<=input;i++))
        do
                for((j=2;j<=input;j++))
                do
                        if [ `expr $i % $j` -eq 0 ]
                                then
                                break
                        fi
                done
                if [ $i -eq $j ]
                then
                        echo $i
                fi
        done
else
        echo "$input is not an integer"
fi
exit 0
